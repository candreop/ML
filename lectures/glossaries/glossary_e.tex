% eigendecomposition
\newglossaryentry{eigendecomposition}{
  name={eigendecomposition},
  description={
    A decomposition of a square matrix $\vect{A}$ into 
    \glspl{eigenvalue} and \glspl{eigenvector}. The matrix $\vect{A}$ 
    can be written as the product $\vect{P}\vect{D}\vect{P}^{-1}$,
    where $\vect{D}$ is a diagonal matrix constructed from the
    \glspl{eigenvalue} of $\vect{A}$, and $\vect{P}$ is an invertible 
    matrix constructed from the \glspl{eigenvector} of $\vect{A}$.
    Several other forms of matrix decomposition exist 
    (e.g. Cholesky decomposition, Hessenberg decomposition, etc)
  }
}
