% Jacobian matrix
\newglossaryentry{Jacobian matrix}{
  name={Jacobian matrix},
  description={
    The Jacobian matrix $J$ of a function 
    $f: \mathbb{R}^n \rightarrow \mathbb{R}^m$,
    is a $m \times n$ matrix of all its first order derivatives
  }
}
