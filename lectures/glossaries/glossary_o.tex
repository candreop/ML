% objective function
\newglossaryentry{objective function}{
  name={objective function},
  description={
    A general term for any function that is optimized during the
    training of a \gls{ml} model
  }
}

% optimisation
\newglossaryentry{optimisation}{
  name={optimisation},
  description={
    A methodology for improving a system, making it as 
    functional and effective as possible.
    In \gls{ml}, optimisation often involves the 
    \gls{extremisation} of an \gls{objective function}
  }
}

% outlier detection
\newglossaryentry{outlier detection}{
  name={outlier detection},
  description={
    Another common term for \gls{anomaly detection}
  }
}

% overfitting
\newglossaryentry{overfitting}{
  name={overfitting},
  description={
    A pathology in the training of a \gls{ml} model,
    characterised by a large gap between the training and test errors
  }
}
