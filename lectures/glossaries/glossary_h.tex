% Hessian matrix
\newglossaryentry{Hessian condition number}{
  name={Hessian condition number},
  description={
    The ratio of the largest \gls{eigenvalue} of a \gls{Hessian matrix}
    to the smallest one.
    It is a measure of how well-conditioned 
    or ill-conditioned the curvature of a function is at a given point
  }
}

% Hessian matrix
\newglossaryentry{Hessian matrix}{
  name={Hessian matrix},
  description={
    The Hessian matrix of a function $f(\vect{x})$
    is a square matrix of second-order partial derivatives 
    $\partial^2 f / \partial x_i \partial x_j$.
    The Hessian matrix is the \gls{Jacobian matrix} of the gradient of $f(\vect{x})$
  }
}
