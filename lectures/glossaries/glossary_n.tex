% negative definite
\newglossaryentry{negative definite}{
  name={negative definite},
  description={
    A symmetric real-valued matrix $\vect{A}$ is negative definite if
    $\vect{x}^T \vect{A} \vect{x}$ is negative for every non-zero real-valued
    column vector $\vect{x}$. All the \glspl{eigenvalue} of a 
    negative definite matrix are negative
  }
}
