% Taylor series
\newglossaryentry{Taylor series}{
  name={Taylor series},
  description={
    The Taylor series (or Taylor expansion) of a function is an approximation
    of that function around point $\vect{x}_0$, 
    expressed as an infinite sum of terms constructed from the 
    derivatives of the function evaluated at $\vect{x}_0$
  }
}
